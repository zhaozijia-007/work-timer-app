<script>

// 初始化图标

lucide.createIcons();

 

// --- 全局配置 ---

const WORK_HOURS = 7;

const WORK_MINUTES = 34;

const WORK_SECONDS = 48;

const STORAGE_KEY = 'userCountdowns';

 

let countdownInterval; // Custom countdown interval

let workCountdownInterval; // Work countdown interval

let workStartTime = 0; // Work start timestamp

let workLeaveTime = 0; // Work leave timestamp

let workTotalDuration = 0; // Total work duration in ms

 

let countdowns = [];

let activeCountdownId = null;

let pendingDeleteId = null; // 用于非阻塞式确认删除

 

// --- 初始化和 Tab 切换 ---

 

window.onload = function() {

const now = new Date();

const hours = String(now.getHours()).padStart(2, '0');

const minutes = String(now.getMinutes()).padStart(2, '0');

document.getElementById('arrivalTime').value = `${hours}:${minutes}`;

loadCountdowns();

renderCountdownList();

 

if (countdowns.length > 0) {

// 如果有倒计时，默认选中第一个

selectCountdown(countdowns[0].id);

}

};

 

function switchTab(tabId) {

// 隐藏所有内容区域

document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));

// 移除所有按钮的激活状态

document.querySelectorAll('.tab-button').forEach(el => el.classList.remove('active'));

 

// 显示选中内容区域

document.getElementById(`content-${tabId}`).classList.remove('hidden');

// 设置选中按钮的激活状态

document.getElementById(`tab-${tabId}`).classList.add('active');

 

if (tabId === 'custom') {

// 切换到自定义倒计时

if (workCountdownInterval) clearInterval(workCountdownInterval); // 停止工作计时

if (activeCountdownId) {

startLiveCountdown(activeCountdownId);

} else {

if (countdownInterval) clearInterval(countdownInterval);

document.getElementById('countdownViewer').classList.add('hidden');

}

} else {

// 切换到工作计算

if (countdownInterval) clearInterval(countdownInterval); // 停止自定义计时

if (workStartTime && workLeaveTime) {

// 如果已经计算过，重新启动工作计时器

startWorkCountdownTicker();

}

}

}

 

// --- 1. 下班时间计算逻辑 (Work Tab) ---

function calculateLeaveTime() {

const inputTime = document.getElementById('arrivalTime').value;

if (!inputTime) return;

 

if (workCountdownInterval) clearInterval(workCountdownInterval); // 停止任何先前的计时器

 

const [inputHours, inputMinutes] = inputTime.split(':').map(Number);

const date = new Date();

date.setHours(inputHours);

date.setMinutes(inputMinutes);

date.setSeconds(0);

date.setMilliseconds(0);

// 记录工作开始时间 (Timestamp)

workStartTime = date.getTime();

 

// 计算预计下班时间 (Date object manipulation)

date.setHours(date.getHours() + WORK_HOURS);

date.setMinutes(date.getMinutes() + WORK_MINUTES);

date.setSeconds(date.getSeconds() + WORK_SECONDS);

date.setMilliseconds(0);

 

// 记录预计下班时间 (Timestamp)

workLeaveTime = date.getTime();

// 计算总工时时长 (ms)

workTotalDuration = workLeaveTime - workStartTime;

 

// 格式化输出

const resultHours = String(date.getHours()).padStart(2, '0');

const resultMinutes = String(date.getMinutes()).padStart(2, '0');

const resultSeconds = String(date.getSeconds()).padStart(2, '0');

 

// 检查是否跨天

const inputDateObj = new Date();

inputDateObj.setHours(inputHours, inputMinutes, 0);

const isNextDay = date.getDate() !== inputDateObj.getDate();

 

// 更新UI

const resultArea = document.getElementById('resultArea');

document.getElementById('leaveTimeDisplay').textContent = `${resultHours}:${resultMinutes}`;

document.getElementById('preciseTimeDisplay').textContent = `精确时间 (含秒): ${resultHours}:${resultMinutes}:${resultSeconds}`;

document.getElementById('nextDayLabel').style.display = isNextDay ? 'block' : 'none';

resultArea.classList.remove('hidden');

resultArea.classList.add('animate-fade-in');

 

// 启动动态工作倒计时和进度条

startWorkCountdownTicker();

}

 

function startWorkCountdownTicker() {

if (workCountdownInterval) clearInterval(workCountdownInterval);

 

function updateWorkTimer() {

const now = new Date().getTime();

let diff = workLeaveTime - now; // 剩余时间

let elapsed = now - workStartTime; // 经过时间

// 1. 工作已完成

if (diff <= 0) {

clearInterval(workCountdownInterval);

updateWorkCountdownDisplay(0, 0, 0, 0);

document.getElementById('progressBar').style.width = '100%';

document.getElementById('progressLabel').textContent = '工作进度: 100.00% (已完成! ��)';

return;

}

 

// 2. 工作尚未开始 (例如：设置的到达时间是未来)

if (elapsed < 0) {

elapsed = 0;

diff = workTotalDuration;

}

// --- 倒计时计算 ---

const days = Math.floor(diff / (1000 * 60 * 60 * 24));

const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));

const seconds = Math.floor((diff % (1000 * 60)) / 1000);

updateWorkCountdownDisplay(days, hours, minutes, seconds);

 

// --- 百分比计算 ---

let percentage = (elapsed / workTotalDuration) * 100;

percentage = Math.min(100, Math.max(0, percentage)).toFixed(2); // 保持在 0-100 之间，保留两位小数

 

document.getElementById('progressBar').style.width = `${percentage}%`;

document.getElementById('progressLabel').textContent = `工作进度: ${percentage}%`;

}

 

function updateWorkCountdownDisplay(d, h, m, s) {

document.getElementById('work-cd-days').textContent = d;

document.getElementById('work-cd-hours').textContent = String(h).padStart(2, '0');

document.getElementById('work-cd-minutes').textContent = String(m).padStart(2, '0');

document.getElementById('work-cd-seconds').textContent = String(s).padStart(2, '0');

}

 

updateWorkTimer(); // 立即执行一次

workCountdownInterval = setInterval(updateWorkTimer, 1000);

}

 

// --- 2. 自定义倒计时逻辑 (Custom Tab) ---

 

function loadCountdowns() {

try {

const stored = localStorage.getItem(STORAGE_KEY);

countdowns = stored ? JSON.parse(stored) : [];

} catch (e) {

console.error("无法加载倒计时数据:", e);

countdowns = [];

}

}

 

function saveCountdowns() {

localStorage.setItem(STORAGE_KEY, JSON.stringify(countdowns));

}

 

function saveNewCountdown() {

const name = document.getElementById('cdName').value.trim();

const targetDate = document.getElementById('cdTargetDate').value;

const errorEl = document.getElementById('cdErrorMessage');

 

if (!name || !targetDate) {

errorEl.textContent = "请填写倒计时名称和目标日期/时间！";

errorEl.className = "text-center text-sm text-red-500 mt-3 h-4 font-bold";

setTimeout(() => errorEl.textContent = '', 3000); // 3秒后清除错误信息

return;

}

 

const newCd = {

id: Date.now().toString(), // 简单ID生成

name: name,

targetTime: new Date(targetDate).getTime()

};

 

countdowns.push(newCd);

saveCountdowns();

renderCountdownList();

// 清空输入框并选中新的倒计时

document.getElementById('cdName').value = '';

document.getElementById('cdTargetDate').value = '';

selectCountdown(newCd.id);

errorEl.textContent = '保存成功！';

errorEl.className = "text-center text-sm text-green-600 mt-3 h-4 font-bold";

setTimeout(() => errorEl.textContent = '', 3000); // 3秒后清除成功信息

}

 

function renderCountdownList() {

const listEl = document.getElementById('countdownList');

// 每次渲染前清空列表

listEl.innerHTML = '';

if (countdowns.length === 0) {

// 如果列表为空，插入提示信息

listEl.innerHTML = '<p class="text-sm text-gray-500 text-center mt-4">暂无倒计时，快创建一个吧！</p>';

document.getElementById('countdownViewer').classList.add('hidden');

return;

}

 

// 修复: 移除之前尝试访问已清除元素的冗余代码

// document.getElementById('emptyListMessage').style.display = 'none'; // 这一行是导致错误的原因，已删除

countdowns.forEach(cd => {

const isActive = cd.id === activeCountdownId;

// 检查是否是待删除状态，用于显示确认按钮

const isPendingDelete = cd.id === pendingDeleteId;

 

const targetDateStr = new Date(cd.targetTime).toLocaleString('zh-CN', {

year: 'numeric', month: 'numeric', day: 'numeric',

hour: '2-digit', minute: '2-digit', hour12: false

});

 

const item = document.createElement('div');

item.className = `flex justify-between items-center p-3 rounded-xl cursor-pointer transition-all ${isActive ? 'bg-rose-100 border border-rose-300 shadow-md' : 'bg-white hover:bg-gray-50 border border-gray-200'}`;

item.onclick = () => selectCountdown(cd.id);

 

// 使用双击确认删除的逻辑替换 confirm()

const deleteButtonHtml = isPendingDelete

? `<button onclick="event.stopPropagation(); deleteCountdown('${cd.id}')" class="text-white bg-red-500 hover:bg-red-600 px-2 py-0.5 rounded-full text-xs transition-colors">确认删除?</button>`

: `<button onclick="event.stopPropagation(); deleteCountdown('${cd.id}')" class="text-gray-400 hover:text-red-500 p-1 rounded-full transition-colors">

<i data-lucide="x" class="w-4 h-4"></i>

</button>`;

 

item.innerHTML = `

<div class="flex flex-col">

<span class="font-semibold ${isActive ? 'text-rose-700' : 'text-gray-800'}">${cd.name}</span>

<span class="text-xs ${isActive ? 'text-rose-500' : 'text-gray-500'}">${targetDateStr}</span>

</div>

${deleteButtonHtml}

`;

listEl.appendChild(item);

lucide.createIcons(); // 重新渲染图标

});

}

 

function deleteCountdown(id) {

// 如果是第二次点击，执行删除

if (pendingDeleteId === id) {

countdowns = countdowns.filter(cd => cd.id !== id);

saveCountdowns();

if (activeCountdownId === id) {

// 如果删除的是当前选中的，则重置选中状态

activeCountdownId = countdowns.length > 0 ? countdowns[0].id : null;

if (countdownInterval) clearInterval(countdownInterval);

document.getElementById('countdownViewer').classList.add('hidden');

}

pendingDeleteId = null;

renderCountdownList();

if (activeCountdownId) {

selectCountdown(activeCountdownId);

}

return;

}

 

// 第一次点击：设置待删除状态并重新渲染列表，显示确认按钮

pendingDeleteId = id;

renderCountdownList();

 

// 5秒后自动清除待删除状态

setTimeout(() => {

if (pendingDeleteId === id) {

pendingDeleteId = null;

renderCountdownList();

}

}, 5000);

}

 

function selectCountdown(id) {

activeCountdownId = id;

pendingDeleteId = null; // 清除待删除状态

renderCountdownList();

startLiveCountdown(id);

}

 

function startLiveCountdown(id) {

const targetCd = countdowns.find(cd => cd.id === id);

const viewerEl = document.getElementById('countdownViewer');

const messageEl = document.getElementById('liveCdMessage');

 

if (!targetCd) {

viewerEl.classList.add('hidden');

return;

}

 

viewerEl.classList.remove('hidden');

document.getElementById('liveCdName').textContent = targetCd.name;

if (countdownInterval) clearInterval(countdownInterval);

 

const targetTime = targetCd.targetTime;

 

function updateTimer() {

const now = new Date().getTime();

const diff = targetTime - now;

 

if (diff < 0) {

clearInterval(countdownInterval);

updateCustomCountdownDisplay(0, 0, 0, 0);

messageEl.textContent = "�� 目标时间已到达！";

messageEl.className = "text-center text-sm text-green-600 mt-3 font-bold";

return;

}

 

const days = Math.floor(diff / (1000 * 60 * 60 * 24));

const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));

const seconds = Math.floor((diff % (1000 * 60)) / 1000);

 

updateCustomCountdownDisplay(days, hours, minutes, seconds);

messageEl.textContent = "";

}

 

function updateCustomCountdownDisplay(d, h, m, s) {

document.getElementById('cd-days').textContent = d;

document.getElementById('cd-hours').textContent = String(h).padStart(2, '0');

document.getElementById('cd-minutes').textContent = String(m).padStart(2, '0');

document.getElementById('cd-seconds').textContent = String(s).padStart(2, '0');

}

 

updateTimer(); // 立即执行一次

countdownInterval = setInterval(updateTimer, 1000);

}
